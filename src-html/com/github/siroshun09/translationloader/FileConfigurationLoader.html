001package com.github.siroshun09.translationloader;
002
003import com.github.siroshun09.configapi.api.file.FileConfiguration;
004import com.github.siroshun09.translationloader.util.LocaleParser;
005import org.jetbrains.annotations.Contract;
006import org.jetbrains.annotations.NotNull;
007
008import java.io.IOException;
009import java.util.Locale;
010
011public class FileConfigurationLoader extends ConfigurationLoader {
012
013    @Contract("_ -> new")
014    public static @NotNull FileConfigurationLoader create(@NotNull FileConfiguration source) {
015        var locale = LocaleParser.fromFileName(source.getPath());
016
017        if (locale != null) {
018            return create(locale, source);
019        } else {
020            throw new IllegalStateException("Could not get the locale");
021        }
022    }
023
024    @Contract("_, _ -> new")
025    public static @NotNull FileConfigurationLoader create(@NotNull Locale locale, @NotNull FileConfiguration source) {
026        return new FileConfigurationLoader(locale, source);
027    }
028
029    private final FileConfiguration source;
030
031    private FileConfigurationLoader(@NotNull Locale locale, @NotNull FileConfiguration source) {
032        super(locale, source);
033        this.source = source;
034    }
035
036    @Override
037    public void load() throws IOException {
038        setLoaded(false);
039
040        try (source) {
041            source.load();
042            super.load();
043        }
044    }
045
046    @Override
047    public void save() throws IOException {
048        if (!isModified()) {
049            return;
050        }
051
052        try (source) {
053            super.save();
054            source.save();
055        }
056    }
057}
