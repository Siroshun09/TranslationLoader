001package com.github.siroshun09.translationloader;
002
003import com.github.siroshun09.configapi.api.file.FileConfiguration;
004import com.github.siroshun09.configapi.api.file.PropertiesConfiguration;
005import com.github.siroshun09.configapi.yaml.YamlConfiguration;
006import net.kyori.adventure.key.Key;
007import net.kyori.adventure.translation.TranslationRegistry;
008import org.jetbrains.annotations.Contract;
009import org.jetbrains.annotations.NotNull;
010
011import java.nio.file.Path;
012import java.util.Locale;
013import java.util.Objects;
014import java.util.function.Function;
015
016/**
017 * A builder of {@link TranslationLoader}.
018 */
019public final class TranslationLoaderBuilder {
020
021    private Path filePath;
022    private Locale locale;
023    private TranslationRegistry registry;
024
025    TranslationLoaderBuilder() {
026    }
027
028    /**
029     * Sets the filepath.
030     *
031     * @param filepath the filepath to load
032     * @return this builder
033     */
034    @Contract(value = "_ -> this", mutates = "this")
035    public @NotNull TranslationLoaderBuilder setFilePath(Path filepath) {
036        this.filePath = filepath;
037        return this;
038    }
039
040    /**
041     * Sets the locale
042     *
043     * @param locale the locale
044     * @return this builder
045     */
046    @Contract(value = "_ -> this", mutates = "this")
047    public @NotNull TranslationLoaderBuilder setLocale(Locale locale) {
048        this.locale = locale;
049        return this;
050    }
051
052    /**
053     * Sets the {@link TranslationRegistry}.
054     *
055     * @param registry the {@link TranslationRegistry}
056     * @return this builder
057     */
058    @Contract(value = "_ -> this", mutates = "this")
059    public @NotNull TranslationLoaderBuilder setRegistry(TranslationRegistry registry) {
060        this.registry = registry;
061        return this;
062    }
063
064    /**
065     * Creates and sets the {@link TranslationRegistry}.
066     *
067     * @param key the {@link Key} of the registry
068     * @return this builder
069     */
070    @Contract(value = "_ -> this", mutates = "this")
071    public @NotNull TranslationLoaderBuilder setRegistry(@NotNull Key key) {
072        this.registry = TranslationRegistry.create(key);
073        return this;
074    }
075
076    /**
077     * Creates new {@link FileConfigurationLoader}.
078     *
079     * @param func the {@link FileConfiguration} creator
080     * @return the new {@link FileConfigurationLoader}
081     */
082    @Contract("_ -> new")
083    public @NotNull FileConfigurationLoader createFileConfigurationLoader(@NotNull Function<Path, FileConfiguration> func) {
084        Objects.requireNonNull(filePath);
085        Objects.requireNonNull(locale);
086        Objects.requireNonNull(registry);
087
088        var config = Objects.requireNonNull(func).apply(filePath);
089        return new FileConfigurationLoader(config, locale, registry);
090    }
091
092    /**
093     * Creates new {@link FileConfigurationLoader} that loads from yaml file.
094     *
095     * @return new {@link FileConfigurationLoader} that loads from yaml file
096     */
097    @Contract(" -> new")
098    public @NotNull FileConfigurationLoader createYamlConfigurationLoader() {
099        return createFileConfigurationLoader(YamlConfiguration::create);
100    }
101
102    /**
103     * Creates new {@link FileConfigurationLoader} that loads from properties file.
104     *
105     * @return new {@link FileConfigurationLoader} that loads from properties file
106     */
107    @Contract(" -> new")
108    public @NotNull FileConfigurationLoader createPropertiesConfigurationLoader() {
109        return createFileConfigurationLoader(PropertiesConfiguration::create);
110    }
111}
