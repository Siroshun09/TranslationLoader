001package com.github.siroshun09.translationloader.argument;
002
003import net.kyori.adventure.text.Component;
004import org.jetbrains.annotations.NotNull;
005
006/**
007 * An interface to create a component with four arguments.
008 *
009 * @param <A1> first argument type
010 * @param <A2> second argument type
011 * @param <A3> third argument type
012 * @param <A4> fourth argument type
013 */
014public interface QuadArgument<A1, A2, A3, A4> {
015
016    /**
017     * Creates a {@link Component}.
018     *
019     * @param a1 first argument
020     * @param a2 second argument
021     * @param a3 third argument
022     * @param a4 fourth argument
023     * @return the {@link Component} that is applied arguments
024     */
025    @NotNull Component apply(@NotNull A1 a1, @NotNull A2 a2, @NotNull A3 a3, @NotNull A4 a4);
026
027}
