001package com.github.siroshun09.translationloader.argument;
002
003import net.kyori.adventure.text.Component;
004import org.jetbrains.annotations.NotNull;
005
006/**
007 * An interface to create a component with one argument.
008 *
009 * @param <A1> the argument type
010 */
011public interface SingleArgument<A1> {
012
013    /**
014     * Creates a {@link Component}.
015     *
016     * @param a1 the argument
017     * @return the {@link Component} that is applied the argument
018     */
019    @NotNull Component apply(@NotNull A1 a1);
020
021}
