001package com.github.siroshun09.translationloader.argument;
002
003import net.kyori.adventure.text.Component;
004import org.jetbrains.annotations.NotNull;
005
006/**
007 * An interface to create a component with two arguments.
008 *
009 * @param <A1> first argument type
010 * @param <A2> second argument type
011 */
012public interface DoubleArgument<A1, A2> {
013
014    /**
015     * Creates a {@link Component}.
016     *
017     * @param a1 first argument
018     * @param a2 second argument
019     * @return the {@link Component} that is applied arguments
020     */
021    @NotNull Component apply(@NotNull A1 a1, @NotNull A2 a2);
022
023}
