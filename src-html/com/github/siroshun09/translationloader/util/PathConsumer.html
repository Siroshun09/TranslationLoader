003import org.jetbrains.annotations.NotNull;
004
005import java.io.IOException;
006import java.nio.file.Path;
007
008/**
009 * A consumer of {@link Path}.
010 */
011@FunctionalInterface
012public interface PathConsumer {
013
014    /**
015     * Performs this operation on the given argument.
016     *
017     * @param path the path
018     * @throws IOException if I/O error occurred
019     */
020    void accept(@NotNull Path path) throws IOException;
021}
