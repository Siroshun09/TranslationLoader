001package com.github.siroshun09.translationloader.util;
002
003import org.jetbrains.annotations.Contract;
004import org.jetbrains.annotations.Nullable;
005
006import java.nio.file.Path;
007import java.util.Locale;
008
009/**
010 * A utility class that parse string to {@link Locale}.
011 */
012public final class LocaleParser {
013
014    private LocaleParser() {
015        throw new UnsupportedOperationException();
016    }
017
018    /**
019     * Parses the {@link Locale} from the file name.
020     *
021     * @param path the path to parse
022     * @return the locale if the parse was successful, null otherwise
023     */
024    @Contract("null -> null")
025    public static @Nullable Locale fromFileName(Path path) {
026        if (path == null) {
027            return null;
028        }
029
030        var filePath = path.getFileName();
031
032        if (filePath == null) {
033            return null;
034        }
035
036        var fileName = filePath.toString().toCharArray();
037        var builder = new StringBuilder();
038
039        for (var c : fileName) {
040            if (c != '.') {
041                builder.append(c);
042            } else {
043                break;
044            }
045        }
046
047        return LocaleParser.parse(builder.toString());
048    }
049
050    /**
051     * Parses string to {@link Locale}
052     *
053     * @param str the string to parse.
054     * @return the locale if the parse was successful, null otherwise
055     */
056    @Contract("null -> null")
057    public static @Nullable Locale parse(String str) {
058        if (str == null || str.isEmpty()) {
059            return null;
060        }
061
062        var segments = str.split("_", 3);
063        int length = segments.length;
064
065        if (length == 1) {
066            return new Locale(str); // language
067        }
068
069        if (length == 2) {
070            return new Locale(segments[0], segments[1]); // language + country
071        }
072
073        if (length == 3) {
074            return new Locale(segments[0], segments[1], segments[2]); // language + country + variant
075        }
076
077        return null;
078    }
079}
