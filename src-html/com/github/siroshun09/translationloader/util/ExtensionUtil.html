001package com.github.siroshun09.translationloader.util;
002
003import org.jetbrains.annotations.NotNull;
004
005import java.nio.file.Path;
006
007/**
008 * A utility class to get file extension.
009 */
010public final class ExtensionUtil {
011
012    private static final String EMPTY_STRING = "";
013    private static final char DOT = '.';
014
015    /**
016     * Gets the extension of file.
017     *
018     * @param path the path to get
019     * @return an extension or an empty string if it could not be obtained
020     */
021    public static @NotNull String getExtension(Path path) {
022        if (path == null) {
023            return EMPTY_STRING;
024        }
025
026        var filePath = path.getFileName();
027
028        if (filePath == null) {
029            return EMPTY_STRING;
030        }
031
032        return getExtension(filePath.toString());
033    }
034
035    /**
036     * Gets the extension of filename.
037     *
038     * @param fileName the filename to get
039     * @return an extension or an empty string if it could not be obtained
040     */
041    public static @NotNull String getExtension(String fileName) {
042        if (fileName == null || fileName.isEmpty()) {
043            return EMPTY_STRING;
044        }
045
046        var lastDot = fileName.lastIndexOf(DOT);
047
048        if (lastDot != -1) {
049            return fileName.substring(lastDot + 1);
050        } else {
051            return EMPTY_STRING;
052        }
053    }
054}
