001package com.github.siroshun09.translationloader.directory;
002
003import com.github.siroshun09.translationloader.TranslationLoader;
004import org.jetbrains.annotations.NotNull;
005import org.jetbrains.annotations.Nullable;
006
007import java.io.IOException;
008import java.util.Locale;
009
010/**
011 * An interface to merge missing messages to translations.
012 */
013@FunctionalInterface
014public interface TranslationLoaderCreator {
015
016    /**
017     * Creates {@link TranslationLoader} of the {@link Locale}.
018     * <p>
019     * The loader is used to compare with the merge target and add missing messages.
020     * <p>
021     * Returning {@link TranslationLoader} must be loaded.
022     *
023     * @param locale the locale of the translation
024     * @return the {@link TranslationLoader} for the locale if it exists, or {@code null} if it does not.
025     * @throws IOException if I/O error occurred
026     */
027    @Nullable TranslationLoader createLoader(@NotNull Locale locale) throws IOException;
028
029}
