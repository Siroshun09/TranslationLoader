001package com.github.siroshun09.translationloader.directory;
002
003import com.github.siroshun09.translationloader.TranslationLoader;
004import com.github.siroshun09.translationloader.util.ExtensionUtil;
005import com.github.siroshun09.translationloader.util.LocaleParser;
006import com.github.siroshun09.translationloader.util.PathConsumer;
007import net.kyori.adventure.key.Key;
008import net.kyori.adventure.translation.GlobalTranslator;
009import net.kyori.adventure.translation.TranslationRegistry;
010import org.jetbrains.annotations.NotNull;
011import org.jetbrains.annotations.Nullable;
012import org.jetbrains.annotations.Unmodifiable;
013
014import java.io.IOException;
015import java.nio.file.Files;
016import java.nio.file.Path;
017import java.util.HashSet;
018import java.util.Objects;
019import java.util.Set;
020import java.util.function.Supplier;
021
022/**
023 * A class that loads messages from files in a directory.
024 */
025public class TranslationDirectory {
026
027    private final Path directory;
028    private final Supplier<TranslationRegistry> registrySupplier;
029
030    private final Set<TranslationLoader> loadedLoaders = new HashSet<>();
031    private TranslationRegistry registry;
032
033    /**
034     * The constructor of {@link TranslationDirectory}.
035     *
036     * @param directory        the directory to load
037     * @param registrySupplier the supplier to create {@link TranslationRegistry}
038     */
039    public TranslationDirectory(@NotNull Path directory,
040                                @NotNull Supplier<TranslationRegistry> registrySupplier) {
041        this.directory = directory;
042        this.registrySupplier = registrySupplier;
043        this.registry = registrySupplier.get();
044    }
045
046    /**
047     * The constructor of {@link TranslationDirectory}.
048     *
049     * @param directory the directory to load
050     * @param key       the key of {@link TranslationRegistry}
051     */
052    public TranslationDirectory(@NotNull Path directory, @NotNull Key key) {
053        this(directory, () -> TranslationRegistry.create(key));
054    }
055
056    /**
057     * Creates the directory if not exists.
058     * <p>
059     * If the directory is created, the {@link PathConsumer} will be called.
060     *
061     * @param onDirectoryCreated {@link PathConsumer}, called when a directory is created.
062     * @throws IOException if I/O error occurred
063     */
064    public void createDirectoryIfNotExists(@NotNull PathConsumer onDirectoryCreated) throws IOException {
065        if (Files.isDirectory(directory)) {
066            return;
067        }
068
069        Files.createDirectories(directory);
070        onDirectoryCreated.accept(directory);
071    }
072
073    /**
074     * Loads message files from directory.
075     *
076     * @throws IOException if I/O error occurred
077     */
078    public void load() throws IOException {
079        if (!loadedLoaders.isEmpty()) {
080            unload();
081        }
082
083        if (!Files.isDirectory(directory)) {
084            return;
085        }
086
087        try (var list = Files.list(directory)) {
088            list.filter(Files::isRegularFile)
089                    .map(this::loadFile)
090                    .filter(Objects::nonNull)
091                    .filter(TranslationLoader::isLoaded)
092                    .forEach(loadedLoaders::add);
093        }
094
095        loadedLoaders.forEach(TranslationLoader::register);
096        GlobalTranslator.get().addSource(registry);
097    }
098
099    /**
100     * Unloads messages from the {@link GlobalTranslator}.
101     */
102    public void unload() {
103        GlobalTranslator.get().removeSource(registry);
104        loadedLoaders.clear();
105        registry = registrySupplier.get();
106    }
107
108    /**
109     * Gets the directory path.
110     *
111     * @return the directory path
112     */
113    public @NotNull Path getDirectory() {
114        return directory;
115    }
116
117    /**
118     * Gets the {@link TranslationRegistry} to register messages
119     *
120     * @return the {@link TranslationRegistry}
121     */
122    public @NotNull TranslationRegistry getRegistry() {
123        return registry;
124    }
125
126    /**
127     * Gets the loaded {@link TranslationLoader}s.
128     *
129     * @return the loaded {@link TranslationLoader}s
130     */
131    public @NotNull @Unmodifiable Set<TranslationLoader> getLoadedTranslationLoaders() {
132        return Set.copyOf(loadedLoaders);
133    }
134
135    /**
136     * Loads messages from the file.
137     *
138     * @param file the file to load
139     * @return the {@link TranslationLoader} of the file
140     */
141    protected @Nullable TranslationLoader loadFile(@NotNull Path file) {
142        var locale = LocaleParser.fromFileName(file);
143
144        if (locale == null) {
145            return null;
146        }
147
148        var builder =
149                TranslationLoader.newBuilder()
150                        .setFilePath(file)
151                        .setLocale(locale)
152                        .setRegistry(registry);
153
154        TranslationLoader loader;
155
156        switch (ExtensionUtil.getExtension(file)) {
157            case "yml":
158            case "yaml":
159                loader = builder.createYamlConfigurationLoader();
160                break;
161            case "properties":
162                loader = builder.createPropertiesConfigurationLoader();
163                break;
164            default:
165                return null;
166        }
167
168        try {
169            loader.load();
170        } catch (IOException e) {
171            throw new RuntimeException(e);
172        }
173
174        return loader;
175    }
176}
