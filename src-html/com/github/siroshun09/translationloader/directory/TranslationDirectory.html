001package com.github.siroshun09.translationloader.directory;
002
003import com.github.siroshun09.configapi.api.file.PropertiesConfiguration;
004import com.github.siroshun09.configapi.yaml.YamlConfiguration;
005import com.github.siroshun09.translationloader.FileConfigurationLoader;
006import com.github.siroshun09.translationloader.TranslationLoader;
007import com.github.siroshun09.translationloader.util.ExtensionUtil;
008import com.github.siroshun09.translationloader.util.LocaleParser;
009import com.github.siroshun09.translationloader.util.PathConsumer;
010import net.kyori.adventure.key.Key;
011import net.kyori.adventure.translation.GlobalTranslator;
012import net.kyori.adventure.translation.TranslationRegistry;
013import org.jetbrains.annotations.Contract;
014import org.jetbrains.annotations.NotNull;
015import org.jetbrains.annotations.Nullable;
016
017import java.io.IOException;
018import java.nio.file.Files;
019import java.nio.file.Path;
020import java.util.HashSet;
021import java.util.Locale;
022import java.util.Objects;
023import java.util.Set;
024import java.util.function.Supplier;
025
026/**
027 * A class that loads messages from files in a directory.
028 */
029public class TranslationDirectory {
030
031    /**
032     * Creates a new {@link TranslationDirectory}.
033     *
034     * @param directory        the directory to load
035     * @param registrySupplier the supplier to create {@link TranslationRegistry}
036     * @return new {@link TranslationDirectory}
037     */
038    @Contract("_, _ -> new")
039    public static @NotNull TranslationDirectory create(@NotNull Path directory,
040                                                       @NotNull Supplier<TranslationRegistry> registrySupplier) {
041        return new TranslationDirectory(directory, registrySupplier, null, null, null);
042    }
043
044    /**
045     * Creates a new {@link TranslationDirectory}.
046     *
047     * @param directory the directory to load
048     * @param key       the key of {@link TranslationRegistry}
049     * @return new {@link TranslationDirectory}
050     */
051    @Contract("_, _ -> new")
052    public static @NotNull TranslationDirectory create(@NotNull Path directory, @NotNull Key key) {
053        return create(directory, () -> TranslationRegistry.create(key));
054    }
055
056    /**
057     * Creates a new {@link TranslationDirectoryBuilder}.
058     *
059     * @return new {@link TranslationDirectoryBuilder}
060     */
061    @Contract(value = " -> new", pure = true)
062    public static @NotNull TranslationDirectoryBuilder newBuilder() {
063        return new TranslationDirectoryBuilder();
064    }
065
066    private final Path directory;
067    private final Supplier<TranslationRegistry> registrySupplier;
068    private final @Nullable PathConsumer onDirectoryCreated;
069    private final @Nullable String version;
070    private final @Nullable TranslationLoaderCreator translationLoaderCreator;
071
072    private final Set<Locale> loadedLocales = new HashSet<>();
073
074    private TranslationRegistry registry;
075
076    TranslationDirectory(@NotNull Path directory, @NotNull Supplier<TranslationRegistry> registrySupplier,
077                         @Nullable PathConsumer onDirectoryCreated,
078                         @Nullable String version, @Nullable TranslationLoaderCreator translationLoaderCreator) {
079        this.directory = directory;
080        this.registrySupplier = registrySupplier;
081        this.onDirectoryCreated = onDirectoryCreated;
082        this.version = version;
083        this.translationLoaderCreator = translationLoaderCreator;
084    }
085
086    /**
087     * Loads message files from directory.
088     * <p>
089     * The process when loading is as follows:
090     * <ol>
091     *     <li>If the {@link TranslationRegistry} already created, call {@link #unload()} to reset it</li>
092     *     <li>If the directory does not exist, create it and call {@link #onDirectoryCreated}</li>
093     *     <li>Get the files in the directory and load them (supports .yml, .yaml, and .properties)</li>
094     *     <li>Update and register loader's messages that {@link TranslationLoader#isLoaded} returns true</li>
095     * </ol>
096     * <p>
097     * Requirements for updating messages (adding missing messages):
098     * <ul>
099     *     <li>Set the {@link TranslationLoaderCreator} by {@link TranslationDirectoryBuilder#setTranslationLoaderCreator(TranslationLoaderCreator)}</li>
100     *     <li>Set the version by {@link TranslationDirectoryBuilder#setVersion(String)}</li>
101     *     <li>The version is not an empty string</li>
102     *     <li>The version is different from {@link TranslationLoader#getVersion} or ends with {@code -SNAPSHOT}</li>
103     *     <li>{@link TranslationLoaderCreator#createLoader(Locale)} returns loaded {@link TranslationLoader}</li>
104     * </ul>
105     *
106     * @throws IOException if I/O error occurred
107     */
108    public void load() throws IOException {
109        if (registry != null) {
110            unload();
111        }
112
113        registry = registrySupplier.get();
114
115        createDirectoryIfNotExists();
116
117        try (var list = Files.list(directory)) {
118            list.filter(Files::isRegularFile)
119                    .map(this::loadFile)
120                    .filter(Objects::nonNull)
121                    .filter(TranslationLoader::isLoaded)
122                    .forEach(this::updateAndRegister);
123        }
124
125        GlobalTranslator.get().addSource(registry);
126    }
127
128    /**
129     * Unloads messages from the {@link GlobalTranslator}.
130     */
131    public void unload() {
132        GlobalTranslator.get().removeSource(registry);
133        loadedLocales.clear();
134        registry = null;
135    }
136
137    /**
138     * Gets the directory path.
139     *
140     * @return the directory path
141     */
142    public @NotNull Path getDirectory() {
143        return directory;
144    }
145
146    /**
147     * Gets the version that this {@link TranslationDirectory} expected to {@link TranslationLoader}.
148     *
149     * @return the version or an empty string if the version is not set
150     */
151    public @NotNull String getVersion() {
152        return Objects.requireNonNullElse(version, "");
153    }
154
155    /**
156     * Gets the {@link TranslationRegistry} to register messages
157     *
158     * @return the {@link TranslationRegistry}
159     */
160    public @NotNull TranslationRegistry getRegistry() {
161        if (registry == null) {
162            throw new IllegalStateException("The registry is not created (Not loaded yet?)");
163        }
164
165        return registry;
166    }
167
168    /**
169     * Gets the set of the loaded {@link Locale}s.
170     *
171     * @return the set of the loaded {@link Locale}s
172     */
173    public Set<Locale> getLoadedLocales() {
174        return loadedLocales;
175    }
176
177    private void createDirectoryIfNotExists() throws IOException {
178        if (!Files.isDirectory(directory)) {
179            Files.createDirectories(directory);
180
181            if (onDirectoryCreated != null) {
182                onDirectoryCreated.accept(directory);
183            }
184        }
185    }
186
187    private @Nullable TranslationLoader loadFile(@NotNull Path file) {
188        var locale = LocaleParser.fromFileName(file);
189
190        if (locale == null) {
191            return null;
192        }
193
194        TranslationLoader loader;
195
196        switch (ExtensionUtil.getExtension(file)) {
197            case "yml":
198            case "yaml":
199                loader = FileConfigurationLoader.create(locale, YamlConfiguration.create(file));
200                break;
201            case "properties":
202                loader = FileConfigurationLoader.create(locale, PropertiesConfiguration.create(file));
203                break;
204            default:
205                return null;
206        }
207
208        try {
209            loader.load();
210        } catch (IOException e) {
211            throw new RuntimeException(e);
212        }
213
214        return loader;
215    }
216
217    private void updateAndRegister(@NotNull TranslationLoader loader) {
218        if (translationLoaderCreator != null && version != null && !version.isEmpty() &&
219                (version.endsWith("-SNAPSHOT") || !loader.getVersion().equals(version))) {
220            TranslationLoader other;
221
222            try {
223                other = translationLoaderCreator.createLoader(loader.getLocale());
224            } catch (IOException e) {
225                throw new RuntimeException("Could not get the merger (" + loader.getLocale() + ")", e);
226            }
227
228            if (other != null && other.isLoaded()) {
229                loader.merge(other);
230                loader.setVersion(other.getVersion());
231
232                try {
233                    loader.save();
234                } catch (IOException e) {
235                    throw new RuntimeException("Could not save the loader", e);
236                }
237            }
238        }
239
240        loader.register(registry);
241        loadedLocales.add(loader.getLocale());
242    }
243}
