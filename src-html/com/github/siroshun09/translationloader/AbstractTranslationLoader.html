001package com.github.siroshun09.translationloader;
002
003import net.kyori.adventure.translation.TranslationRegistry;
004import org.jetbrains.annotations.NotNull;
005import org.jetbrains.annotations.Unmodifiable;
006
007import java.nio.file.Path;
008import java.text.MessageFormat;
009import java.util.Collections;
010import java.util.HashMap;
011import java.util.Locale;
012import java.util.Map;
013import java.util.Objects;
014import java.util.stream.Collectors;
015
016/**
017 * An abstract implementation of {@link TranslationLoader}.
018 */
019public abstract class AbstractTranslationLoader implements TranslationLoader {
020
021    private final Path path;
022    private final Locale locale;
023    private final TranslationRegistry registry;
024    private final Map<String, String> messageMap = new HashMap<>();
025
026    private boolean isLoaded = false;
027
028    /**
029     * The constructor of {@link AbstractTranslationLoader}.
030     *
031     * @param path     the filepath to load
032     * @param locale   the locale of messages that this loader will load
033     * @param registry the {@link TranslationRegistry} to register messages
034     */
035    protected AbstractTranslationLoader(@NotNull Path path, @NotNull Locale locale,
036                                        @NotNull TranslationRegistry registry) {
037        this.path = Objects.requireNonNull(path);
038        this.locale = Objects.requireNonNull(locale);
039        this.registry = Objects.requireNonNull(registry);
040    }
041
042    @Override
043    public boolean isLoaded() {
044        return isLoaded;
045    }
046
047    /**
048     * Set if the message was loaded successfully.
049     *
050     * @param isLoaded true if the message was loaded, false otherwise
051     */
052    protected void setLoaded(boolean isLoaded) {
053        this.isLoaded = isLoaded;
054    }
055
056    @Override
057    public boolean register() {
058        if (!isLoaded) {
059            return false;
060        }
061
062        var messageFormatMap =
063                messageMap.entrySet()
064                        .stream()
065                        .collect(
066                                Collectors.toUnmodifiableMap(
067                                        Map.Entry::getKey,
068                                        entry -> new MessageFormat(entry.getValue())
069                                )
070                        );
071
072        registry.registerAll(locale, messageFormatMap);
073
074        return true;
075    }
076
077    @Override
078    public @NotNull Path getFilePath() {
079        return path;
080    }
081
082    @Override
083    public @NotNull TranslationRegistry getRegistry() {
084        return registry;
085    }
086
087    @Override
088    public @NotNull Locale getLocale() {
089        return locale;
090    }
091
092    @Override
093    public @NotNull @Unmodifiable Map<String, String> getMessageMap() {
094        return Collections.unmodifiableMap(messageMap);
095    }
096
097    /**
098     * Gets the modifiable message map.
099     *
100     * @return the modifiable message map
101     */
102    protected @NotNull Map<String, String> getModifiableMessageMap() {
103        return messageMap;
104    }
105}
